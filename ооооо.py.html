import flet as ft
from flet import icons  # Ensure you import icons correctly
import matplotlib.pyplot as plt
from io import BytesIO
import base64

class DataApp:
    def __init__(self, page: ft.Page):
        self.page = page
        self.page.title = "Приложение для Статистики Питания"
        self.page.horizontal_alignment = ft.CrossAxisAlignment.CENTER
        self.page.vertical_alignment = ft.MainAxisAlignment.CENTER

        # Initialize theme
        self.page.theme_mode = "light"
        self.light_theme = {
            "background": "#E0FFF1",  # Light pistachio background
            "text": "#000000",  # Black text
            "button": "#FFFFFF",  # White buttons
            "button_text": "#000000"  # Black button text
        }
        self.dark_theme = {
            "background": "#2E0249",  # Dark purple background
            "text": "#FFFFFF",  # White text
            "button": "#4B0082",  # Indigo buttons
            "button_text": "#FFFFFF"  # White button text
        }

        self.data = []
        self.users = {}
        self.current_user = None
        self.diagrams = []
        self.current_diagram_index = 0
        self.selected_date = None
        self.selected_meal_type = None
        self.product_list = [
            {"name": "Яблоко", "unit": "г", "proteins": 0.3, "fats": 0.2, "carbs": 14, "calories": 52},
        ]
        self.daily_norms = {
            "calories": 2000,
            "proteins": 50,
            "fats": 70,
            "carbs": 300
        }
        self.create_widgets()

    def apply_theme(self):
        theme = self.light_theme if self.page.theme_mode == "light" else self.dark_theme
        self.page.theme = ft.Theme(color_scheme_seed=theme["background"])
        self.page.bgcolor = theme["background"]
        if self.page.controls:
            self.page.controls[0].bgcolor = theme["background"]
            self.page.controls[0].controls[0].color = theme["text"]
            self.page.controls[0].controls[1].color = theme["text"]
            self.page.controls[0].controls[2].color = theme["text"]
            self.page.controls[0].controls[3].bgcolor = theme["button"]
            self.page.controls[0].controls[3].color = theme["button_text"]
            self.page.controls[0].controls[4].bgcolor = theme["button"]
            self.page.controls[0].controls[4].color = theme["button_text"]
            self.page.controls[0].controls[5].color = theme["text"]

    def create_widgets(self):
        self.create_login_widgets()
        self.create_main_widgets()
        self.show_login_widgets()

    def create_login_widgets(self):
        self.login_frame = ft.Column(
            controls=[
                ft.TextField(label="Имя", on_submit=self.login_on_enter),
                ft.TextField(label="Фамилия", on_submit=self.login_on_enter),
                ft.TextField(label="Пароль", password=True, on_submit=self.login_on_enter),
                ft.ElevatedButton(text="Войти", on_click=self.login),
                ft.ElevatedButton(text="Зарегистрироваться", on_click=self.register),
                ft.Text(value="", color=ft.Colors.RED),
                ft.IconButton(icon=icons.LIGHT_MODE, on_click=self.toggle_theme)
            ],
            horizontal_alignment=ft.CrossAxisAlignment.CENTER,
            width=300
        )

    def create_main_widgets(self):
        self.main_frame = ft.Column(
            controls=[
                ft.Row(
                    controls=[
                        ft.Text(value="Дата:", style="headlineSmall"),
                        ft.DatePicker(on_change=self.on_date_select),
                    ],
                    alignment=ft.MainAxisAlignment.CENTER
                ),
                ft.Row(
                    controls=[
                        ft.Text(value="Прием пищи:", style="headlineSmall"),
                        ft.Dropdown(
                            options=[
                                ft.dropdown.Option(text="Завтрак"),
                                ft.dropdown.Option(text="Обед"),
                                ft.dropdown.Option(text="Ужин")
                            ],
                            on_change=self.add_meal_entry
                        ),
                    ],
                    alignment=ft.MainAxisAlignment.CENTER
                ),
                ft.Row(
                    controls=[
                        ft.Text(value="Продукт:", style="headlineSmall"),
                        ft.Dropdown(
                            options=[ft.dropdown.Option(text=product["name"]) for product in self.product_list],
                            on_change=self.on_combobox_select
                        ),
                    ],
                    alignment=ft.MainAxisAlignment.CENTER
                ),
                ft.Row(
                    controls=[
                        ft.Text(value="Количество:", style="headlineSmall"),
                        ft.TextField(on_submit=self.add_data_on_enter),
                    ],
                    alignment=ft.MainAxisAlignment.CENTER
                ),
                ft.Row(
                    controls=[
                        ft.Text(value="Единица измерения:", style="headlineSmall"),
                        ft.Dropdown(
                            options=[
                                ft.dropdown.Option(text="шт"),
                                ft.dropdown.Option(text="г"),
                                ft.dropdown.Option(text="мл"),
                                ft.dropdown.Option(text="л"),
                                ft.dropdown.Option(text="ст.л."),
                                ft.dropdown.Option(text="ч.л.")
                            ],
                            on_change=self.select_unit
                        ),
                    ],
                    alignment=ft.MainAxisAlignment.CENTER
                ),
                ft.ElevatedButton(text="Добавить данные", on_click=self.add_data),
                ft.ElevatedButton(text="Показать статистику", on_click=self.show_statistics),
                ft.ElevatedButton(text="Очистить данные", on_click=self.clear_data),
                ft.ElevatedButton(text="Выйти", on_click=self.logout),
                ft.ListView(expand=1, spacing=10, padding=20, auto_scroll=True),
                ft.Text(value="", color=ft.Colors.RED),
                ft.Row(
                    controls=[
                        ft.Text(value="Нормы:", style="headlineSmall"),
                        ft.TextField(label="Калории", value=str(self.daily_norms["calories"])),
                        ft.TextField(label="Белки", value=str(self.daily_norms["proteins"])),
                        ft.TextField(label="Жиры", value=str(self.daily_norms["fats"])),
                        ft.TextField(label="Углеводы", value=str(self.daily_norms["carbs"])),
                        ft.ElevatedButton(text="Сохранить нормы", on_click=self.save_norms),
                    ],
                    alignment=ft.MainAxisAlignment.CENTER
                ),
                ft.Row(
                    controls=[
                        ft.IconButton(icon=icons.ARROW_BACK, on_click=self.show_previous_diagram),
                        ft.IconButton(icon=icons.ARROW_FORWARD, on_click=self.show_next_diagram),
                    ],
                    alignment=ft.MainAxisAlignment.CENTER
                ),
                ft.Image(src="", width=800, height=600),
                ft.IconButton(icon=icons.LIGHT_MODE, on_click=self.toggle_theme)
            ],
            horizontal_alignment=ft.CrossAxisAlignment.CENTER,
            width=1000
        )

    def show_login_widgets(self):
        self.page.views.clear()
        self.page.views.append(ft.View("/login", controls=[self.login_frame]))
        self.page.go("/login")

    def show_main_widgets(self):
        self.page.views.clear()
        self.page.views.append(ft.View("/main", controls=[self.main_frame]))
        self.page.go("/main")
        self.apply_theme()  # Apply theme after creating main widgets

    def login(self, e):
        name = self.login_frame.controls[0].value
        surname = self.login_frame.controls[1].value
        password = self.login_frame.controls[2].value

        if name and surname and password:
            user_key = f"{name}_{surname}"
            if user_key in self.users and self.users[user_key] == password:
                self.current_user = user_key
                self.show_main_widgets()
                self.login_frame.controls[4].value = ""
            else:
                self.login_frame.controls[4].value = "Неверные учетные данные"
                self.login_frame.controls[4].color = ft.Colors.RED
        else:
            self.login_frame.controls[4].value = "Пожалуйста, заполните все поля"
            self.login_frame.controls[4].color = ft.Colors.RED

    def login_on_enter(self, e):
        self.login(e)

    def register(self, e):
        name = self.login_frame.controls[0].value
        surname = self.login_frame.controls[1].value
        password = self.login_frame.controls[2].value

        if name and surname and password:
            user_key = f"{name}_{surname}"
            if user_key not in self.users:
                self.users[user_key] = password
                self.login_frame.controls[4].value = "Регистрация успешна!"
                self.login_frame.controls[4].color = ft.Colors.GREEN
            else:
                self.login_frame.controls[4].value = "Пользователь уже существует"
                self.login_frame.controls[4].color = ft.Colors.RED
        else:
            self.login_frame.controls[4].value = "Пожалуйста, заполните все поля"
            self.login_frame.controls[4].color = ft.Colors.RED

    def logout(self, e):
        self.current_user = None
        self.data = []
        self.main_frame.controls[9].controls.clear()
        self.main_frame.controls[8].value = "Количество данных: 0"
        self.show_login_widgets()

    def add_data(self, e):
        product_name = self.main_frame.controls[2].controls[1].value
        quantity = self.main_frame.controls[3].controls[1].value
        unit = self.main_frame.controls[4].controls[1].value

        if product_name and quantity and unit:
            try:
                quantity = float(quantity)
                product_info = next((p for p in self.product_list if p["name"] == product_name), None)
                if product_info:
                    base_unit = product_info["unit"]
                    if base_unit == unit:
                        proteins = product_info["proteins"] * quantity
                        fats = product_info["fats"] * quantity
                        carbs = product_info["carbs"] * quantity
                        calories = product_info["calories"] * quantity
                    else:
                        self.main_frame.controls[10].value = "Несовпадение единиц измерения"
                        self.main_frame.controls[10].color = ft.Colors.RED
                        return
                    self.data.append((self.selected_date, self.selected_meal_type, product_name, proteins, fats, carbs, calories))
                    self.main_frame.controls[3].controls[1].value = ""
                    self.update_data_listbox()
                    self.main_frame.controls[10].value = "Данные успешно добавлены!"
                    self.main_frame.controls[10].color = ft.Colors.GREEN
                else:
                    self.main_frame.controls[10].value = "Продукт не найден"
                    self.main_frame.controls[10].color = ft.Colors.RED
            except ValueError:
                self.main_frame.controls[10].value = "Пожалуйста, введите корректное количество."
                self.main_frame.controls[10].color = ft.Colors.RED
        else:
            self.main_frame.controls[10].value = "Пожалуйста, заполните все поля."
            self.main_frame.controls[10].color = ft.Colors.RED

    def add_data_on_enter(self, e):
        self.add_data(e)

    def delete_selected_data(self, e):
        selected_indices = self.main_frame.controls[9].selected_indices
        if selected_indices:
            for index in sorted(selected_indices, reverse=True):
                del self.data[index]
            self.update_data_listbox()
            self.main_frame.controls[10].value = "Данные успешно удалены!"
            self.main_frame.controls[10].color = ft.Colors.GREEN

    def update_data_listbox(self):
        self.main_frame.controls[9].controls.clear()
        for item in self.data:
            self.main_frame.controls[9].controls.append(
                ft.Text(f"{item[2]}: Белки={item[3]:.2f}, Жиры={item[4]:.2f}, Углеводы={item[5]:.2f}, Калории={item[6]:.2f}")
            )
        self.main_frame.controls[8].value = f"Количество данных: {len(self.data)}"

    def show_statistics(self, e):
        if not self.data:
            self.main_frame.controls[10].value = "Нет данных для отображения статистики."
            self.main_frame.controls[10].color = ft.Colors.RED
            return

        proteins, fats, carbs, calories = self.calculate_statistics()

        fig, ax = plt.subplots(figsize=(8, 6), dpi=100)
        ax.bar(['Белки', 'Жиры', 'Углеводы', 'Калории'], [proteins, fats, carbs, calories / 10], color=['blue', 'green', 'red', 'purple'])
        ax.set_ylabel('Количество')
        ax.set_title('Статистика питания')

        # Add norms and differences
        norm_proteins = float(self.main_frame.controls[11].controls[1].value)
        norm_fats = float(self.main_frame.controls[11].controls[2].value)
        norm_carbs = float(self.main_frame.controls[11].controls[3].value)
        norm_calories = float(self.main_frame.controls[11].controls[4].value) / 10

        diff_proteins = proteins - norm_proteins
        diff_fats = fats - norm_fats
        diff_carbs = carbs - norm_carbs
        diff_calories = (calories / 10) - norm_calories

        ax.axhline(y=norm_proteins, color='blue', linestyle='--', label=f'Норма белков: {norm_proteins} г')
        ax.axhline(y=norm_fats, color='green', linestyle='--', label=f'Норма жиров: {norm_fats} г')
        ax.axhline(y=norm_carbs, color='red', linestyle='--', label=f'Норма углеводов: {norm_carbs} г')
        ax.axhline(y=norm_calories, color='purple', linestyle='--', label=f'Норма калорий: {norm_calories * 10} ккал')

        ax.legend()

        buf = BytesIO()
        fig.savefig(buf, format='png')
        buf.seek(0)
        img_str = base64.b64encode(buf.read()).decode('utf-8')
        self.main_frame.controls[13].src = f"data:image/png;base64,{img_str}"

        # Display differences in the label
        self.main_frame.controls[10].value = f"Разница от нормы: Белки={diff_proteins:.2f} г, Жиры={diff_fats:.2f} г, Углеводы={diff_carbs:.2f} г, Калории={diff_calories * 10:.2f} ккал"
        self.main_frame.controls[10].color = ft.Colors.BLACK

    def show_current_diagram(self):
        for widget in self.main_frame.controls[12].controls:
            if isinstance(widget, ft.IconButton):
                continue
            widget.visible = False
        if self.diagrams:
            self.diagrams[self.current_diagram_index].visible = True

    def show_previous_diagram(self, e):
        if self.current_diagram_index > 0:
            self.current_diagram_index -= 1
            self.show_current_diagram()

    def show_next_diagram(self, e):
        if self.current_diagram_index < len(self.diagrams) - 1:
            self.current_diagram_index += 1
            self.show_current_diagram()

    def calculate_statistics(self):
        proteins = sum(item[3] for item in self.data)
        fats = sum(item[4] for item in self.data)
        carbs = sum(item[5] for item in self.data)
        calories = sum(item[6] for item in self.data)
        return proteins, fats, carbs, calories

    def clear_data(self, e):
        self.data = []
        self.main_frame.controls[9].controls.clear()
        self.main_frame.controls[8].value = "Количество данных: 0"
        self.main_frame.controls[10].value = "Данные успешно очищены!"
        self.main_frame.controls[10].color = ft.Colors.GREEN
        self.diagrams = []
        self.current_diagram_index = 0
        self.show_current_diagram()

    def update_combobox_values(self, e):
        # Получаем текущий ввод пользователя
        input_text = self.main_frame.controls[2].controls[1].value.lower()

        # Фильтруем список продуктов на основе введенного текста
        filtered_products = [product for product in self.product_list if input_text in product["name"].lower()]

        # Сортируем продукты: сначала те, которые начинаются с введенного текста, затем остальные
        filtered_products.sort(key=lambda x: (not x["name"].lower().startswith(input_text), x["name"].lower()))

        # Обновляем значения в Combobox
        self.main_frame.controls[2].controls[1].options = [ft.dropdown.Option(text=product["name"]) for product in filtered_products]

    def on_combobox_select(self, e):
        # Обработка выбора элемента из Combobox
        selected_product = self.main_frame.controls[2].controls[1].value
        # Здесь можно добавить дополнительную логику при выборе продукта

    def select_unit(self, e):
        self.selected_unit = e.control.value

    def on_date_select(self, e):
        self.selected_date = e.control.value

    def add_meal_entry(self, e):
        self.selected_meal_type = e.control.value
        self.main_frame.controls[10].value = f"Выбран прием пищи: {self.selected_meal_type}"
        self.main_frame.controls[10].color = ft.Colors.BLACK

    def save_norms(self, e):
        try:
            calories_norm = float(self.main_frame.controls[11].controls[1].value)
            proteins_norm = float(self.main_frame.controls[11].controls[2].value)
            fats_norm = float(self.main_frame.controls[11].controls[3].value)
            carbs_norm = float(self.main_frame.controls[11].controls[4].value)

            self.daily_norms["calories"] = calories_norm
            self.daily_norms["proteins"] = proteins_norm
            self.daily_norms["fats"] = fats_norm
            self.daily_norms["carbs"] = carbs_norm

            self.main_frame.controls[10].value = "Нормы успешно сохранены!"
            self.main_frame.controls[10].color = ft.Colors.GREEN
        except ValueError:
            self.main_frame.controls[10].value = "Пожалуйста, введите корректные значения для норм."
            self.main_frame.controls[10].color = ft.Colors.RED

    def toggle_theme(self, e):
        self.page.theme_mode = "light" if self.page.theme_mode == "dark" else "dark"
        self.apply_theme()

if __name__ == "__main__":
    ft.app(target=DataApp)
